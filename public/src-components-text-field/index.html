Basic usage

Name
<div style={{ padding: 40 }}>
  <Formik
    initialValues={{ name: '' }}
    onSubmit={values => console.log(values)}
    enableReinitialize
  >
    {({ isSubmitting }) => (
      <Form>
        <TextField name="name" label="Name" />
      </Form>
    )}
  </Formik>
</div>
